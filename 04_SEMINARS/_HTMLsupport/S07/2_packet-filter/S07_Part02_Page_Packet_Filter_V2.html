def parse_ethernet_header(data):
dest_mac, src_mac, proto = struct.unpack("! 6s 6s H", data[:14])
return dest_mac, src_mac, proto
 
def parse_ipv4_header(data):
version_ihl = data[0]
version = version_ihl >> 4
ihl = (version_ihl & 0x0F) * 4
ttl, proto, src, dst = struct.unpack("! 8x B B 2x 4s 4s", data[:20])
return src, dst, proto
 
def passes_filter(src_ip, dst_ip, proto, src_port, dst_port):
# Rule 1: Source Network
if not src_ip.startswith("10."):
return False
 
# Rule 2: DNS & Safe TCP
if proto == 17 and dst_port == 53:
return True
if proto == 6 and dst_port > 1024:
return True
 
# Rule 3: Allow all TCP
if proto == 6:
return True
 
return False
